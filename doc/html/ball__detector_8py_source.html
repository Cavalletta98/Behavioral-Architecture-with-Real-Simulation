100  np_arr = np.fromstring(ros_data.data, np.uint8)
101  image_np = cv2.imdecode(np_arr, cv2.IMREAD_COLOR) # OpenCV >= 3.0:
102 
103  greenLower = (50, 50, 20)
104  greenUpper = (70, 255, 255)
105 
106  blurred = cv2.GaussianBlur(image_np, (11, 11), 0)
107  hsv = cv2.cvtColor(blurred, cv2.COLOR_BGR2HSV)
108  mask = cv2.inRange(hsv, greenLower, greenUpper)
109  mask = cv2.erode(mask, None, iterations=2)
110  mask = cv2.dilate(mask, None, iterations=2)
111  #cv2.imshow('mask', mask)
112  cnts = cv2.findContours(mask.copy(), cv2.RETR_EXTERNAL,
113  cv2.CHAIN_APPROX_SIMPLE)
114  cnts = imutils.grab_contours(cnts)
115  center = None
116  # only proceed if at least one contour was found
117  if len(cnts) > 0:
118  # find the largest contour in the mask, then use
119  # it to compute the minimum enclosing circle and
120  # centroid
121  c = max(cnts, key=cv2.contourArea)
122  ((x, y), radius) = cv2.minEnclosingCircle(c)
123  M = cv2.moments(c)
124  center = (int(M["m10"] / M["m00"]), int(M["m01"] / M["m00"]))
125 
126  # only proceed if the radius meets a minimum size
127  if radius > 10:
128  # draw the circle and centroid on the frame,
129  # then update the list of tracked points
130  cv2.circle(image_np, (int(x), int(y)), int(radius),
131  (0, 255, 255), 2)
132  cv2.circle(image_np, center, 5, (0, 0, 255), -1)
133 
134  self.resp_center = center[0]
135  self.resp_radius = radius
136 
137  else:
138  self.resp_center = -1
139  self.resp_radius = -1
